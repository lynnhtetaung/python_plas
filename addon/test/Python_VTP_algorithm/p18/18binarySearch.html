#A quick demo of Binary Search

def binarySearch(alist, item):
    
    first = 0
    last = len(alist)-1
    found = False
    
    while first<=last and not found:
        midpoint = (first + last)//2
        if alist[midpoint] == item:
            found = True
        else:
            if item < alist[midpoint]:
                last = midpoint-1
                print("What is last value", last)
            else:
                first = midpoint+1
                print("What is first value", first)
	
    return found
	
	
testlist = [1, 2, 8, 13, 48]
print(binarySearch(testlist, 3))
print(binarySearch(testlist, 13))
