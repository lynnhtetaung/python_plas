#A quick demo of Sequential Search

def sequentialSearch(alist, item):
    pos = 0
    found = False
	
    while pos < len(alist) and not found:
        if alist[pos] == item:
            found = True
        else:
            pos = pos+1
            print(pos)
            
    return found

testlist = [1, 2, 32, 8]
print(sequentialSearch(testlist, 8))
print(sequentialSearch(testlist, 13))
