# A quick-demo of using stack operations

class Stack:
    def __init__(self):
        self.items = []

    def isEmpty(self):
        return self.items == []

    def push(self, item):
        self.items.append(item)

    def pop(self):
        return self.items.pop()

    def peek(self):
        return self.items[len(self.items)-1]

    def size(self):
        return len(self.items)
   
s=Stack()    
print("Is the stack empty?", s.isEmpty())
s.push(4)
s.push('dog')
print("What is the peek element?", s.peek())
s.push(True)
print("The size of the stack is", s.size())
print("Is the stack empty?", s.isEmpty())
s.push(8.4)
print("The element for the first pop operation is", s.pop())
print("The element for the second pop operation is", s.pop())
print("The size of the stack is", s.size())
