# A quick-demo of using queue operations

class Queue:
    def __init__(self):
        self.items = []
	
    def isEmpty(self):
        return self.items == []
	
    def enqueue(self, item):
        self.items.insert(0,item)
	
    def dequeue(self):
        return self.items.pop()

    def size(self):
        return len(self.items)
	
q=Queue()	
print("Is the queqe empty?", q.isEmpty())
q.enqueue(4)
print("The element for the first dequeue operation is ", q.dequeue())
q.enqueue('dog')
q.enqueue(True)
print("The size of the queue is", q.size())

q.enqueue(8.4)
print("The element for the second dequeue operation is ", q.dequeue())
