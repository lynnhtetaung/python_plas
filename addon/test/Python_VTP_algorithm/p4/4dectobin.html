# A quick-demo of Converting Decimal Numbers to Binary Numbers using stack

from pythonds.basic import Stack
    
def divideBy2(decNumber):
    remstack = Stack()
    binString1 = ""
                           
    while decNumber > 0:
        rem = decNumber % 2
        remstack.push(rem)
        binString1 = binString1 + str(remstack.peek())
        decNumber = decNumber // 2
        print("The decimal number is", decNumber)
                               
    print("The push data in the stack is", binString1)
                  
    binString2 = ""
    while not remstack.isEmpty():
        binString2 = binString2 + str(remstack.pop())
                
    return binString2
               
print("The binary number for decimal value(42) is:", divideBy2(42))
