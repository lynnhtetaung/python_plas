# A quick-demo of Converting Decimal Numbers to any Based Numbers using stack

from pythonds.basic import Stack
    
def baseConverter(decNumber,base):
     digits = "0123456789ABCDEF"
     anyString1 = ""
            
     remstack = Stack()
            
     while decNumber > 0:
          rem = decNumber % base
          remstack.push(rem)
          anyString1 = anyString1 + str(remstack.peek())
          decNumber = decNumber // base
                    
     print("The push data in the stack is", anyString1)
            
     newString = ""
     while not remstack.isEmpty():
          newString = newString + digits[remstack.pop()]
            
     return newString
            
print("The first converting value is:", baseConverter(88,3))
print("The second converting value is:", baseConverter(30,16))
