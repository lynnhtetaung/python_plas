# A quick-demo of hybrid inheritance
# Creating a Base class named Company:
class Company:
  def __init__(self):
    # Initializing a class variable named comp to store company name:
    self.comp = "Royal"
  def display(self):  # Method to print the Company Name:
    print("The company name is: ",self.comp)

# 1st Derived or Child Class of University Class:
class Department(Company):
  def __init__(self):     
    Company.__init__(self)
    self.dept = "IT"
  def display(self):  # Method to print the Department Name:
    # using "class" name to access display method defined in the parent class:
    print("The department name is: ",self.dept)
    Company.display(self)

# 2nd Derived or Child Class of Company Class:
class Branch(Company):
  def __init__(self):    
    self.branch = "Data Science"
  def display(self):  # Method to print the Branch Name:
    print("The branch name is: ",self.branch)

# Derived or Child Class of Class Course and Branch:
class Employee(Department, Branch):
  def __init__(self):    
    self.name = "Brown"
    Branch.__init__(self)
    Department.__init__(self)
  def display(self):
    print("The name of the employee is: ",self.name)
    Branch.display(self)
    Department.display(self)
    
# Object Instantiation:
ob = Employee()  # Object named ob of the class Employee.
ob.display()    # Calling the display method of Employee class.
