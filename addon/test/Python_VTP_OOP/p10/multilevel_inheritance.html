# A quick-demo of multilevel inheritance
# Base class
class Main_Company:

	def __init__(self, companyname):
		self.companyname = companyname

# Intermediate class
class Branch_Company(Main_Company):
	def __init__(self, branchcompanyname, companyname):
		self.branchcompanyname = branchcompanyname

		# invoking constructor of Company class
		Main_Company.__init__(self, companyname)

# Derived class
class Partnership(Branch_Company):
	def __init__(self,partnername, branchcompanyname, companyname):
		self.partnername = partnername

		# invoking constructor of Branch_Company class
		Branch_Company.__init__(self, branchcompanyname, companyname)

	def print_name(self):
		print('Main company name :', self.companyname)
		print("Branch company name :", self.branchcompanyname)
		print("Partnership company name :", self.partnername)

# Driver code
p1 = Partnership('Champion', 'Grand', 'Solutions Sign')
print(p1.companyname)
p1.print_name()
