# A quick-demo of multiple inheritance
# Base class1
class Mother:
	mothername = ""
	def mother(self):
		print(self.mothername)

# Base class2
class Father:
	fathername = ""
	def father(self):
		print(self.fathername)

# Derived class
class Daugher(Mother, Father):
	def parents(self):
		print("Father :", self.fathername)
		print("Mother :", self.mothername)

# Driver's code
d1 = Daugher()
d1.fathername = "RAM"
d1.mothername = "SITA"
d1.parents()
