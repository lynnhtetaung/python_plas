# A quick-demo of single inheritance
# Base class
class Vehicle:
    def Vehicle_info(self):
        print('Inside Vehicle class')

# Child class
class Flight(Vehicle):
    def flight_info(self):
        print('Inside flight class')

# Driver's code
# Create object of Flight
flight = Flight()
# access Vehicle's info using Flight object
flight.Vehicle_info()
flight.flight_info()
