# A quick-demo of using getter and setter method to implement encapsulation properly
class Employee:
    def __init__(self, name, salary):
        self.name = name
       # private member
        self.__salary = salary

    # getter method
    def get_salary(self):
        return self.__salary

    # setter method
    def set_salary(self, salary):
        self.__salary = salary

emp = Employee('Jessa', 100000)

# retrieving salary using getter
print('Name:', emp.name, emp.get_salary())

# changing age using setter
emp.set_salary(200000)

# retrieving age using getter
print('Name:', emp.name, emp.get_salary())
