# A quick-demo of accessing protected members that can be accessible within the class and also available to its sub-classes(encapsulation)
# base class
class University:
    def __init__(self):
        # Protected member
        self._department = "IT"

# child class
class Teacher(University):
    def __init__(self, name):
        self.name = name
        University.__init__(self)

    def show(self):
        print("Teacher name :", self.name)
        # Accessing protected member in child class
        print("Working on department :", self._department)

t = Teacher("Merry")
t.show()
# Direct access protected data member
print('Department:', t._department)
