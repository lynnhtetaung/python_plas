# A quick-demo of hierarchical inheritance
# Base class
class Electronic_Device:
    def info(self):
        print("This is electronic device.")

# First child class
class Laptop(Electronic_Device):
    def laptop_info(self, name):
        print("Laptop name is:", name)

# Second child class
class Phone(Electronic_Device):
    def phone_info(self, name):
        print("Phone name is:", name)

# Driver code
obj1 = Laptop()
obj1.info()
obj1.laptop_info('HP')

obj2 = Phone()
obj2.info()
obj2.phone_info('Samsung')
