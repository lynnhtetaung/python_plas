# A quick-demo of accessing private members via public instance methods (encapsulation)
class Student:
    # constructor
    def __init__(self, name, major):
        # public data member
        self.name = name
        # private member
        self.__major = major

    # public instance methods
    def show(self):
        # private members are accessible from a class
        print("Name:",self.name, 'Major:',self.__major)

# creating object of a student class
stu = Student('John', 'IT')

# calling public method of the class
stu.show()
