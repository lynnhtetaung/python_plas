# A quick-demo of constructor with default values
class Employee:
    # constructor with default values age and job
    def __init__(self, name, age=34, job='IT'):
        self.name = name
        self.age = age
        self.job = job

    # display Employee
    def show(self):
        print(self.name, self.age, self.job)

# creating object of the Employee class
e1 = Employee('Merry')
e1.show()

rossy = Employee('Rossy', 25)
rossy.show()
