# A quick-demo of accessing private members via name mangling(encapsulation)
class Student:
    # constructor
    def __init__(self, name, major):
        # public data member
        self.name = name
        # private member
        self.__major = major

# creating object of a Student class
stu = Student('Jessa', 'Business')

print('Name:', stu.name)
# direct access to private member using name mangling
print('Major:', stu._Student__major)
