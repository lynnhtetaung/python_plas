# A quick-demo of deconstructor usage to delete the objects
class Student:

    # constructor
    def __init__(self, name):
        print('Inside Constructor!')
        self.name = name
        
    def show(self):
        print('My name is', self.name)

    # destructor
    def __del__(self):
        print('Object is destroyed within destructor!')
        
# create object
s1 = Student('Emma')
s1.show()

# delete object
del s1
