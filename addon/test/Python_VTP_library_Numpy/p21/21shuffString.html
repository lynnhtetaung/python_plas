# A quick-demo of shuffling string 
# The random.shuffle() doesn?t? work with String. And so, shuffle a string can be done using following step by step approaches.
# Convert String to list
# Shuffle the list randomly
# Convert the shuffled list into String

import random

sampleStr = "Hello"
print("Original String: ", sampleStr)

# convert string into list
char_list = list(sampleStr)

# shuffle list
random.shuffle(char_list)

# convert list to string
finalStr = ''.join(char_list)

# Assume the resulted shuffled string is Hleol
print("shuffled String is: ", finalStr)
