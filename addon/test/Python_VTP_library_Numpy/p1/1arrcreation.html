# A quick-demo of Array Creation on Numpy

import numpy as np
 
# Creating a rank 1 Array and the output will be printed like [4 7 9]
arr = np.array([1, 2, 3])
print("Array with Rank 1: \n",arr)
 
# Creating a rank 2 Array and the output will be printed like [[4 7 8]
                                                              [5 3 2]]
arr = np.array([[1, 2, 3],
                [4, 5, 6]])
print("Array with Rank 2: \n", arr)
 
# Creating an array from tuple and the output will be printed like [4 7 9]
arr = np.array((1, 3, 2))
print("Array created using "
      "passed tuple:\n", arr)

# Creating array from list with type float 
a = np.array([[2, 4], [8, 7]], dtype = 'float') 
print ("Array created using passed list:\n", a) 

# Creating a 1X2 array with all zeros 
c = np.zeros((1, 2)) 
print ("\nAn array initialized with all zeros:\n", c) 

# Create a sequence of integers from 0 to 30 with steps of 5 
f = np.arange(0, 30, 5) 
print ("\nA sequential array with steps of 5:\n", f) 

# Reshaping 2X4 array to 2X2X2 array 
arr = np.array([[1, 2, 3, 4], 
               [1, 2, 0, 1]]) 
newarr = arr.reshape(2, 2, 2) 
print ("Reshaped array:\n", newarr) 

# Flatten array 
arr = np.array([[1, 2, 3], [4, 5, 6]]) 
flarr = arr.flatten() 

print ("Original array:\n", arr) 
print ("Fattened array:\n", flarr)
