# A quick-demo of string Comparision usage on Numpy
# numpy.equal() function: This function checks for string1 == string2 elementwise.
# numpy.not_equal(): This function checks whether two string is unequal or not.
# numpy.greater(): This function checks whether string1 is greater than string2 or not.
# numpy.greater_equal(): It checks whether string1 >= string2 or not.
# numpy.less_equal() 	It checks whether string1 is <= string2 or not.
# numpy.less() 	It check whether string1 is lesser than string2 or not.
 
import numpy as np
 
# comparing a string elementwise using equal() method
a=np.char.equal('geeks','for')
 
print("'geek' and 'for' are the equal:",a)

# comparing a string elementwise using not_equal() method
a=np.char.not_equal('geeks','for')
 
print("'geek' and 'for' are not the equal:",a)

# comparing a string elementwise using greater() method
a=np.char.greater('geeks','for')
 
print("'geek' is greater than 'for':",a)

a  = np.greater_equal([8., 2.], [5., 3.]) 
print("Greater_Equal :", a)   

a = np.array([4,2])

print("Is a greater_equal than b :", a >= b)

a  = np.less_equal([8., 2.], [5., 3.]) 
print("Less_Equal :", a)
  
a  = np.less([8., 2.], [5., 3.]) 
print("Less :", a) 
  
a = np.array([4,2])

print("Is a lesser than b :", a < b)
