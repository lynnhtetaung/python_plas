# A quick-demo of numpy.arange method usage
# arange returns evenly spaced values within a given interval. step size is specified.
# arange([start,] stop[, step,][, dtype]) : Returns an array with evenly spaced elements as per the interval. 

import numpy as np

# create the array with values from 0 to 4
array = np.arange(5)

# The output will be printed like this format [8 7 2 3 5]
print("First array : \n", array, "\n") 
 
# if np.arrange(3, 7), then the output will be printed like this [3 4 5 6] 
print("Second array :\n", np.arange(4, 10), "\n")
 
# if np.arrange(3, 9, 3), then the output will be printed like this [3 6]
print("Third array:\n", np.arange(4, 10, 3), "\n")

# if np.arrange(3, 9, 3), then the output will be printed like this [ 3.  6.]
print("Fourth array:\n", np.arange(4, 10, 2, float), "\n")
