# A quick-demo of Arithmetic Functions usage
# divide() function: Array element from first array is divided by elements from second element
# multiply() function: Multiply arguments element-wise.
# numpy.add() function: Add arguments element-wise. 
# numpy.subtract() function: Subtract arguments element-wise. 
# numpy.negative() function: Numerical negative, element-wise.
# power() function: First array elements raised to powers from second array, element-wise.

import numpy as np
 
# input_array
arr1 = [2, 27, 2, 21, 23]
arr2 = [2, 3, 4, 5, 6]
print ("arr1 : ", arr1)
print ("arr2 : ", arr2)
 
# output_array for division
out = np.divide(arr1, arr2)
print ("\nOutput array after division: \n", out)

# output_array for multiplication
out = np.multiply(arr1, arr2)
print ("\nOutput array after multiplication: \n", out)
