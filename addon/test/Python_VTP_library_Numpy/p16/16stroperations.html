# A quick-demo of String Operations on NumPy
# numpy.lower() function:This function returns the lowercase string from the given string.
# numpy.upper() function:This function returns the uppercase string from the given string.
# numpy.split() function:This function returns a list of strings after breaking the given string by the specified separator.
# numpy.join() function:This function is a string method and returns a string in which the elements of sequence have been joined by str separator.
 
import numpy as np
 
# converting to lowercase
print("Converting to lowercase", np.char.lower('HELLO'))
 
# converting to uppercase
print("Converting to uppercase", np.char.upper('hello'))

# splitting a string
print("Splitting a string", np.char.split('geeks for geeks'))
 
# splitting a string
print("Another format for splitting a string",np.char.split('geeks, for, geeks', sep = ','))

# Joining a string
print("Joining a string with str sperator",np.char.join('-', 'hello'))
 
# Joining a string
print("Another format for joining by str sperator",np.char.join(['-', ':'], ['geeks', 'for']))
