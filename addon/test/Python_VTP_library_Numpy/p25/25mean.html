# A quick-demo of numpy.mean() method
# numpy.mean(arr, axis = None) : Compute the arithmetic mean (average) of the given data (array elements) along the specified axis.
# axis : axis along which we want to calculate the arithmetic mean. axis = 0 means along the column and axis = 1 means working along the row.

import numpy as np 
	
# 1D array 
arr = [20, 2, 7, 1, 34] 

print("arr : ", arr) 

# if arr = [2, 3, 4, 5],then (2+3+4+5)/4 = 3.5 and the output will be printed like 3.5.
print("mean of arr : ", np.mean(arr)) 

# 2D array 
arr = [[4, 1, 0], 
       [5, 6, 2], 
       [3, 2, 4]] 
	
# mean of the flattened array, calculate the sum of all values and then divided by 9
print("\nmean of arr, axis = None : ", np.mean(arr)) 
	
# mean along the axis = 0 that calculates mean value along each column
print("\nmean of arr, axis = 0 : ", np.mean(arr, axis = 0)) 

# mean along the axis = 1 that calculates mean value along each row
print("\nmean of arr, axis = 1 : ", np.mean(arr, axis = 1))
