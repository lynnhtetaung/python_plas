# A quick-demo of numpy.dot() method 
  
import numpy as np 
  
# Scalars : in np.dot( 2,3), the output will be 6.
product = np.dot(5, 4) 
print("Dot Product of scalar values :", product) 
  
''' Vectors : in vector_a = 3 + 4j and vector_b = 2 + 5j,
the dot product 
= 3(2 + 5j) + 4j(2 ? 5j)
= 6 + 15j + 8j ? 20
= -14 + 23j
'''
vector_a = 2 + 3j
vector_b = 4 + 5j

product = np.dot(vector_a, vector_b) 
print("Dot Product of scalar values :", product) 
  
# 2D array 
arr_a = np.array([[1, 4], [5, 6]]) 
arr_b = np.array([[2, 4], [5, 2]]) 
  
product = np.dot(arr_a, arr_b) 
print("Dot Product in 2D array  : \n", product)
