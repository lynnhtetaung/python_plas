# A quick-demo of basic array characteristics in Numpy

import numpy as np 
  
# Creating array object 
arr = np.array( [[ 1, 2, 3], 
                 [ 4, 2, 5]] ) 
  
 
# Printing array dimensions (axes) 
print("No. of dimensions: ", arr.ndim) 
  
# Printing shape of array and the output will be printed like this (1, 2), 1 means for row and 2 means for column
print("Shape of array: ", arr.shape) 
  
# Printing size (total number of elements) of array 
print("Size of array: ", arr.size) 
  
# Printing type of elements in array and the output will print like this int32 or float64
print("Array stores elements of type: ", arr.dtype)
