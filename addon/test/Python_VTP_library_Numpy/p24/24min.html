# A quick-demo of numpy.minimum() method 
# numpy.minimum() function is used to find the element-wise minimum of array elements.
# It compares two arrays and returns a new array containing the element-wise minima.
# If one of the elements being compared is a NaN, then that element is returned. If both elements are NaNs then the first is returned.
  
import numpy as np

in_num1 = 10
in_num2 = 21

print ("Input number1 : ", in_num1) 
print ("Input number2 : ", in_num2) 
	
out_num = np.minimum(in_num1, in_num2) 
print ("minimum of 10 and 21:", out_num) 

in_arr1 = [2, 8, 125] 
in_arr2 = [3, 3, 15] 

print ("Input array1 : ", in_arr1) 
print ("Input array2 : ", in_arr2) 
	
out_arr = np.minimum(in_arr1, in_arr2)

# if in_arr1 = [1, 3, 128] and in_arr2 = [0, 7, 15], the output will be printed like [ 0  3 15] 
print ("Output array after selecting minimum: ", out_arr)

in_arr1 = [np.nan, 0, np.nan] 
in_arr2 = [np.nan, np.nan, 0] 
   
print ("Input array1 : ", in_arr1)  
print ("Input array2 : ", in_arr2) 
    
out_arr = np.minimum(in_arr1, in_arr2)  
print ("Output array after selecting minimum: ", out_arr)
