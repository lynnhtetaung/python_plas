# A quick-demo of iterating over numpy array
#The order of iteration is chosen to match the memory layout of an array, 
#without considering a particular ordering.This can be seen by iterating over the transpose of the array.
 
import numpy as np
 
# creating an array using arrange method
a = np.arange(12)
 
# shape array with 3 rows and 4 columns
a = a.reshape(3,4)
 
print('Original array is:')
print(a)
print()
 
print('Modified array is:')
 
# iterating  an array
for x in np.nditer(a):
    print(x, end = ",")

# Transpose of original array
b = a.T 

print('\nTranspose array is: \n',b)
        
print('Print the values fron transpose array is:')
for x in np.nditer(b): 
    print(x, end = ",")
