# A quiclk-demo of basic operations on single array

import numpy as np
 
# Defining Array 1
a = np.array([[1, 2],
              [3, 4]])
 
# Defining Array 2
b = np.array([[4, 2],
              [2, 8]])
               
# Adding 1 to every element
print ("Adding 1 to every element:\n", a + 1)
 
# Subtracting 2 from each element
print ("Subtracting 2 from each element:\n", b - 2)

# Multiplying 3 to every element
print ("Multiplying 3 to every element:\n", a * 3)

# Dividing 2 from each element
print ("Dividing 2 to every element:\n", b / 2)
 
# sum of array elements Performing Unary operations
print ("Sum of all array elements: ", a.sum())
 
# Adding two arrays Performing Binary operations
print ("Array sum:\n", a + b)

# square each element 
print ("Squaring each element:\n", a**2)
