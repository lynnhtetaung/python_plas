# A quick demo of separator and end keyword usage
#The sep key is used for adding separators into the string provided.
#The end key will set the string that needs to be appended when printing is done.

# sperator keywoard usage
print('hello', 'world', sep=None)
print('hello', 'world', sep=' ')
print('hello', 'world', sep='')
print('hello', 'world', sep='\n')
print('home', 'user', 'documents', sep='/')
print('/home', 'user', 'documents', sep='/')
print('node', 'child', 'child', sep=' -> ')

# end keywoard usage
print('Checking file integrity...', end='')
print('ok')
print('The first sentence', end='. ')
print('The second sentence', end='. ')
print(end= '\n')

# mix usage of the two keyword arguments(sep and end)
print('Mercury', 'Venus', sep=', ', end=', ')
print('Mars', 'Jupiter',sep=', ')
