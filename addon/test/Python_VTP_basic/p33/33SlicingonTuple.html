# A quick-demo of Slicing operation in tuples

my_data = (11, 22, 33, 44, 55, 66)
print(my_data)

# elements from 3rd to 4th
print(my_data[2:4])

# elements from start to 3th
print(my_data[:3])

# elements from 5th to end
print(my_data[4:])

# elements from 4th to second last
print(my_data[3:-1])
