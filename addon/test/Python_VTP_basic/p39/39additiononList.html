# A quick-demo of addition on List 

# list of numbers
n_list = [1, 2, 3]

# 1. adding item at the desired location
# adding element 100 at the third location
n_list.insert(2, 100)

# list: [1, 2, 100, 3]
print(n_list)

# 2. adding element at the end of the list
n_list.append(99)

# list: [1, 2, 100, 3, 99]
print(n_list)

# 3. adding several elements at the end of list
# the following statement can also be written like this:
# n_list + [11, 22]
n_list.extend([11, 22])

# list: [1, 2, 100, 3, 99, 11, 22]
print(n_list)
