# A quick-demo of getting a sublist on List using slicing

# list of numbers
n_list = [1, 2, 3, 4, 5, 6, 7]

# Whole list
print(n_list[:])

# list items from 2nd to 3rd
print(n_list[1:3])

# list items from beginning to 3rd
print(n_list[:3])

# list items from 4th to end of list
print(n_list[3:])
