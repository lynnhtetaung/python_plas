# A quick-demo of mumpy.argsort()
# numpy.argsort():This function returns the indices that would sort an array.


import numpy as np
 
# Numpy array created
a = np.array([9, 3, 1, 7, 4, 3, 6])
 
# unsorted array print
print('Original array:\n', a)
 
# Sort array indices
b = np.argsort(a)
print('Sorted indices of original array->', b)
 
# To get sorted array using sorted indices
# c is temp array created of same len as of b
c = np.zeros(len(b), dtype = int)
for i in range(0, len(b)):
    c[i]= a[b[i]]
print('Sorted array->', c)

The output of this program is as follows:
Original array:
 [9 3 1 7 4 3 6]
Sorted indices of original array-> [2 1  4 6  0]
Sorted array-> [1  3 4 6 7 ]
