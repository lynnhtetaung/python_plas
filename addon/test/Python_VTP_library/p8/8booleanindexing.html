# A quick-demo of boolean array indexing usage
#Boolean Array Indexing:Elements are returned which satisfy that Boolean expression.
#It is used for filtering the desired element values.

import numpy as np
 
a = np.array([10, 40, 80, 50, 100])

# Select numbers greater than 50
print("Select the numbers grater than 50 :",a[a>50])

a = np.array([10, 40, 80, 50, 100])

# print the squaring to the multiples of 40
# if a[a%50==0]**2, the output will be printed like [2500 10000]

print("Squaring the multiples of 40 :",a[a%40==0]**2)
