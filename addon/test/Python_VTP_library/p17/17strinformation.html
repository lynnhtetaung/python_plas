# A quick-demo of String Informatin Usage
# numpy.count() function:This function returns the number of occurrences of a substring in the given string.
# numpy.rfind() : This function returns the highest index of the substring if found in given string. If not found then it returns -1.
# numpy.isnumeric() : This function returns ?True? if all characters in the string are numeric characters, Otherwise, It returns ?False?.
 
import numpy as np
 
a=np.array(['geeks', 'for', 'geeks'])
 
# counting a substring and the output will be printed like [0 0 1]
print("Counting a substring 'geek' in array a:",np.char.count(a,'geek'))
 
# counting a substring and the output will be printed like [0 0 1]
print("Counting a substring 'fo' in array a:",np.char.count(a, 'fo'))

# Finding a substring and the output will be printed like [0 -1 0]
print("Finding a substring 'geek' in array a:",np.char.rfind(a,'geek'))
 
# counting a substring and the output will be printed like [0 -1 0]
print("Finding a substring 'fo' in array a:",np.char.rfind(a, 'fo'))

# Checking numeric or not and the ouput will be printed as True or False
print("Is 'geeks' numeric?",np.char.isnumeric('geeks'))
 
# Checking numeric or not and the ouput will be printed as True or False
print("Is '12' numeric?",np.char.isnumeric('12'))
