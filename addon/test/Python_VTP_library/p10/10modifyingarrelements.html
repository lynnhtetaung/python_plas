# A quick-demo of modifying array values by using op_flags 
#Its default value is read-only, but can be set to read-write or write-only mode. This will enable modifying array elements using the iterator nditer.
 
import numpy as np
 
# creating an array using arrange 
# method
a = np.arange(12)
 
# shape array with 3 rows and 
# 4 columns 
a = a.reshape(3,4)
print('Original array is:')
print(a)
print()
 
# modifying array values
for x in np.nditer(a, op_flags = ['readwrite']):
    x[...] = 5*x
print('Modified array is:')
print(a)
