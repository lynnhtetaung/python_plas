# A quick-demo of Broadcasting Iteration on numpy array
# If two arrays are broadcastable, a combined nditer object is able to iterate upon them concurrently.
 
import numpy as np
 
# creating an array using arrange 
# method 
a = np.arange(12)
 
 # shape array with 3 rows and 
# 4 columns 
a = a.reshape(3,4) 
 
print('First array is:') 
print(a)
print()  
 
# Creating second array using 
# array method
print('Second array is:') 
b = np.array([5, 6, 7, 8], dtype = int) 
print(b)  
print() 
 
print('Modified array is:')
for x,y in np.nditer([a,b]): 
    print("%d:%d" % (x,y), end = ",")
