# A quick-demo of silicing of elements in a Array
  
# importing array module 
import array as arr 
  
# creating a list  
l = [1, 2, 3, 4, 5] 
  
a = arr.array('i', l) 
print("Intial Array: ") 
for j in (a): 
    print(j, end =",") 
  
# Print elements of a range using Slice operation 
Sliced_array = a[1:3] 
print("\nSlicing elements in a range 1-3: ") 
print(Sliced_array) 
  
# Print elements from a pre-defined point to end 
Sliced_array = a[2:] 
print("\nElements sliced from 2nd "
      "element till the end: ") 
print(Sliced_array) 
  
# Printing elements from beginning till end 
Sliced_array = a[:] 
print("\nPrinting all elements using slice operation: ") 
print(Sliced_array)
