# A quick-demo of adding elements to an array 
# A new element can be added at the beginning, end, or any given index of array using insert() function. append() is used to add the value the end of the array.
  
# importing "array" for array creations 
import array as arr 
  
# array with int type 
a = arr.array('i', [1, 2, 3]) 
  
  
print ("Array before insertion ", end =": ") 
for j in range (0, 3): 
    print (a[j], end =",") 
print() 
  
# inserting array using insert() function 
a.insert(1, 4) 
  
print ("Array after insertion ", end =": ") 
for j in (a): 
    print (j, end =",") 
print()  
  
# array with float type 
b = arr.array('d', [2.5, 3.2, 3.3]) 
  
print ("Array before insertion ", end =": ") 
for i in range (0, 3): 
    print (b[i], end =",") 
print() 
  
# adding an element using append() 
b.append(4.4) 
  
print ("Array after insertion ", end =": ") 
for i in (b): 
    print (i, end =",") 
print()
