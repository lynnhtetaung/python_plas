# A quick-demo of json file reading usage for each category

''' Assume keypoint.json includes as follow:
{
	"version":1.1,
	"people":[
		{
			"pose_keypoints":[
				17.7,71.6,0.92,74.8
			],			
                          "face_keypoints":[             
				6,7,8,9
			],
		}
	]
} '''

import json

def json_read(file_name):
   
    with open(file_name) as f:
        
        dict_data = json.load(f)
        for people in dict_data['people']:
            print (people['pose_keypoints'])
            
        for people in dict_data['people']:
            print (people['face_keypoints'])            
    
if __name__ == "__main__":
    people = json_read("keypoint.json")
