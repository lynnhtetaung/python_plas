# A quick-demo of Updating a json string
    
import json 
   
# JSON data: 
x_json = '{"name": "Jame", "languages": ["Japanese", "Chinese"]}'
  
# python object to be appended 
y_dict = {"pin":110096} 
  
# parsing JSON string: 
z_json = json.loads(x_json) 
   
# appending the data 
z_json.update(y_dict) 
  
# the result is a JSON string: 
print(json.dumps(z_json))
