# A quick demo of CSV file reading
''' Assume testCSV.csv includes the following data:
           City          State  Capital      Population
0  Philadelphia   Pennsylvania       No   1.581 Million
1    Sacramento     California      Yes     0.5 Million
2      New York       New York       No   8.623 Million
3        Austin          Texas      Yes    0.95 Million
4         Miami        Florida       No   0.463 Million '''

import pandas as pd

# read the csv file
titanic_data = pd.read_csv(r'D:\testCSV.csv')

col_names = ['Id',
             'Survived',
             'Passenger Class',
             'Full Name',
             ]
# default column headers is the first row of a CSV file. However, custom header names can be set by putting header = None as follows:
titanic_data = pd.read_csv(r'D:\testCSV.csv', names=col_names, header=None)
print(titanic_data)

# data reading from the line specified by header
df_header_2 = pd.read_csv(r'D:\testCSV.csv', header=2)
print(df_header_2)

# data reading by selecting necessary columns
df = pd.read_csv(r'D:\testCSV.csv',index_col=0, usecols=[0, 1, 3])
print(df) 

# data reading by selecting necessary rows
df = pd.read_csv(r'D:\testCSV.csv', index_col=0, skiprows=range(1, 5, 2))
print(df)
