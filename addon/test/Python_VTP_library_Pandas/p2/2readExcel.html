# A quick demo of data reading from excel file
''' Assume the Datafile.xlsx includes the following data:
For Sheet1,
        A   B   C
one    11  12  13
two    21  22  23
three  31  32  33 and
For Sheet2,
        A   B   C
one    77  88  99
two    21  22  23
three  31  32  33 '''

import pandas as pd

# read excel file
df = pd.read_excel('D:\Datafile.xlsx')

# Specify the sheet to read with the argument sheet_name by specifing the number (starting at 0)
df_sheet_index = pd.read_excel('D:\Datafile.xlsx', sheet_name=0)
print(df_sheet_index)

# Specify the sheet by defining sheet name.
df_sheet_name = pd.read_excel('D:\Datafile.xlsx', sheet_name='sheet2')
print(df_sheet_name)

# load multiple data by specifying a number of 0 starting or the sheet name.
df_sheet_multi = pd.read_excel('D:\Datafile.xlsx', sheet_name=[0, 'sheet2'])

# all sheets are read by specifying sheet_name argument is none.
df_sheet_all = pd.read_excel('D:\Datafile.xlsx', sheet_name=None)
