# A quick-demo of changing the default pandas values: the followings are default parameters
#display.max_rows and display.max_columns which shows the default number of rows and columns.
#display.max_colwidth which gives us the maximum width of the column
#display.expand_frame_repr which gives us DataFrames that is spread across numerous pages.
#display.precision gives us the precision of the decimal numbers

import pandas as pd

#.get_option define parameter with a particular detail about the default values in pandas.

df1= pd.get_option("display.max_rows")
print("The maximum number of rows ", df1)

df2= pd.get_option("display.max_columns")
print("The maximum number of columns ",df2)

# .set_option() allows to change a default value to desired something.
pd.set_option("display.max_columns",10)
df3= pd.get_option("display.max_columns")
print("The updated maximum number of columns ",df3)

# .reset_option can get back the default values which may change previously.
pd.reset_option("display.max_columns")
df4= pd.get_option("display.max_columns")
print("The current maximum number of rows ", df4)

#.describe_option describes the parameter with details
pd.describe_option("display.max_rows")
