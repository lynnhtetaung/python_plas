# A quick demo of creating Date Time object by using .date_range function with the following parameter:
#First parameter: start= 'dd/mm/yyyy'.
#Second parameter: periods= n, where n is no of periods or date time elements you need.
#Third parameter: freq= 'x', where 'x' can be 'H'(hour), 'D'(days), 'W'(weeks), 'M'(month), 'Y'(years), etc.

import pandas as pd

# print date in yyyy-mm-dd format
df_rng = pd.DataFrame() 
df_rng['date'] = pd.date_range('1/1/2011', periods = 72, freq ='H')
print(df_rng[:2])

# breaking the time and date into separate features
df_rng['year'] = df_rng['date'].dt.year  #creates 'year' column and extracts year 
df_rng['day'] = df_rng['date'].dt.day  #creates 'day' column and extracts day
df_rng['hour'] = df_rng['date'].dt.hour #creates 'hour' column and extracts hour
print(df_rng.head(3))

# to get the present time as a timestamp
df_time = pd.Timestamp.now()
print(df_time)
