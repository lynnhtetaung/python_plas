# A quick demo of indexing and selection on pandas

import pandas as pd

# n dimentional array into a pandas dataframe
data = [['Axel',32], ['Alice', 26], ['Alex', 45]]
df = pd.DataFrame(data,columns=['Name','Age'])
print(df)

# to select a column, the column name can be used.
print(df['Name'])

# to add a column to a dataframe.
c = pd.DataFrame([1,2,3], columns=['Example'])
df['Example'] = c['Example']
print(df)

# delete a column using the keyword del.
del df['Example']

# select a row using .loc[label].
print(df.loc[0])

# another method to select a row using .iloc[label].
print(df.iloc[2])

# delete the first row. 
df = df.drop(0)
print(df)

# .append() appends the row on the dataframe.
user = pd.DataFrame([['Vivian',33]], columns= ['Name','Age'])

# add it to the existing dataframe.
df = df.append(user)
print(df)
