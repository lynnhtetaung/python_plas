# A quick demo of data frame creation
import pandas as pd

# a single list into a pandas dataframe
data = [1,2,3]
df = pd.DataFrame(data)
print(df)

# n dimentional array into a pandas dataframe
data = [['Axel',32], ['Alice', 26], ['Alex', 45]]
df = pd.DataFrame(data,columns=['Name','Age'])
print(df)

# from dictionary into a pandas dataframe
d = {'one':[1,2,3], 'two':[2,3,4], 'three':[3,4,5] }
df = pd.DataFrame(d)
print(df)

# from dataframe into a dataframe by copying parts of a dataframe into a new dataframe.
df1 = df[['one','two']].copy()
print(df1)
