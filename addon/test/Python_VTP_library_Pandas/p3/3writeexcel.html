# A quick demo of writing data to excel file
import pandas as pd
import openpyxl

# create data frame
df = pd.DataFrame([[11, 21, 31], [12, 22, 32], [31, 32, 33]],
                  index=['one', 'two', 'three'], columns=['a', 'b', 'c'])
df.to_excel('D:\DataWrite.xlsx', index=True, header=False)
print(df)

# Write multiple DataFrames to Excel files
df2 = df[['a', 'c']]
print(df2)
with pd.ExcelWriter('D:\pandas_to_excel.xlsx') as writer:
    df.to_excel(writer, sheet_name='sheet1')
    df2.to_excel(writer, sheet_name='sheet2')
