# A quick-demo of iteration on pandas
#iteritems(): Helps to iterate over each element of the set, column-wise.
#iterrows(): Each element of the set, row-wise.
#itertuple(): Each row and form a tuple out of them.
''' Assume the csv file includes the following data:
         City           State         Capital      
0    Philadelphia    Pennsylvania      No 
1    Sacramento      California        Yes   
2    New York         New York         No   
 '''
import pandas as pd

#load a dataset 
df = pd.read_csv(r'D:\testCSV.csv')

# travel and visit each and every value of the dataset.
for key,values in df.iteritems():
    print(key, values)
    
# visit all the elements of a dataset, row-wise.
for row_index,row in df.iterrows():
    print(row_index, row)

# creates a tuple for every row in the dataset and give a tuple of the rows present.
for row in df.itertuples():
    print(row)
