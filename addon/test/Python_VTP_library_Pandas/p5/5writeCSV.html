# A quick-demo of writing CSV
import pandas as pd
import csv

# create data frame
city = pd.DataFrame([['Sacramento', 'California'], ['Miami', 'Florida']], columns=['City', 'State'])

# data writing to csv file
city.to_csv('D:\city.csv')

# print the csv file
print(city)

# another way to write as csv file
myData = [[1, 2, 3], ['Good Morning', 'Good Evening', 'Good Afternoon']]
myFile = open('D:\Greeting.csv', 'w')
with myFile:
   writer = csv.writer(myFile)
   writer.writerows(myData)
